weights = вес (В начем случае это время необходимое на выполнение одной работы)
er_time -> early run time = раннее время выполнения
lr_time -> later run time = позднее время выполнения
count_s -> count sort = отдельный счётчик для проверки каждой работы 
count_len -> count length = счётчик длины (нужен для формирования итогово массива)
fork = вилка (развилка для подсчёта количества разветвлений в сетевом графе)
path = дорожка
lenf -> length fork = длина дорожки
count_f -> count fork = счётчик для развилки
ways = пути
transition = переход (последний переход в пути"path")
number_wn -> number work new = временная переменная, работает в связке с переменной transition 
brk -> break = остановка (считает сколько путей ещё осталось пройти)
path_quantity = количество путей
worst_case = наихудший вариант (функция)
choose_path = выбрать путь
probability_success = вероятность успеха (функция)
disp_select -> variance selective = выборочная дисперсия
t_avg -> time average = среднее
